// Define the pin for the Buzzer
const int buzzerPin = 3;  // Connect the Buzzer to Digital Pin 9
// Melody notes
int melody[] = {262, 294, 330, 349, 392, 440, 494, 523}; // C4 to C5
int noteDuration = 500; // Duration of each note in milliseconds

void setup() {
  // Set the buzzer pin as an output
  pinMode(buzzerPin, OUTPUT);
}

void loop() {
  for (int i = 0; i < 8; i++) {
    tone(buzzerPin, melody[i]); // Play the note
    delay(noteDuration);         // Wait for the note duration
    noTone(buzzerPin);           // Stop sound
    delay(50);                   // Brief pause between notes
  }
  //delay(1000); // Wait for 1 second before repeating
  for (int i = 7; i >= 0; i--) {
    tone(buzzerPin, melody[i]); // Play the note
    delay(noteDuration);         // Wait for the note duration
    noTone(buzzerPin);           // Stop sound
    delay(50);                   // Brief pause between notes
  }
  //delay(1000); // Wait for 1 second before repeating
  
}
